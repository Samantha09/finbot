```ts
import type { AnyAgentTool } from "../types.js";
import { toToolResult } from "../types.js";
import { fetchGfEtfList } from "./gf-etf-search.js";

const EtfRotationStrategySchema = {
  type: "object" as const,
  properties: {
    symbols: {
      type: "array" as const,
      items: { type: "string" as const },
      description: "ETF 代码列表，如 [\"510050.SH\", \"159915.SZ\"]",
    },
    period: {
      type: "string" as const,
      enum: ["short", "medium", "long"],
      description: "轮动周期：short=短期(1月内) medium=中期(3-6月) long=长期(1年+)",
    },
    maxResults: {
      type: "integer" as const,
      description: "返回数量上限，默认全部",
    },
  },
  required: ["symbols", "period"],
};

interface EtfData {
  tradeCode: string;
  secuAbbr?: string;
  pePercent?: number;
  pbPercent?: number;
  roc1m?: number;
  roc3m?: number;
  roc6m?: number;
  netMainForce5d?: number;
  netMainForce10d?: number;
  assetScale?: number;
  sharpRatio1y?: number;
  sharpRatio3y?: number;
}

export function calculateMomentumScore(data: Pick<EtfData, "roc1m" | "roc3m" | "roc6m">, period: string): number {
  const { roc1m = 0, roc3m = 0, roc6m = 0 } = data;
  let raw: number;
  if (period === "short") {
    raw = roc1m * 0.5 + roc3m * 0.3 + roc6m * 0.2;
  } else if (period === "medium") {
    raw = roc3m * 0.5 + roc6m * 0.3 + roc1m * 0.2;
  } else {
    raw = roc6m * 0.5 + roc3m * 0.3 + roc1m * 0.2;
  }
  // Normalize: assume -20%~+20% as full range
  return Math.min(100, Math.max(0, (raw + 20) * 2.5));
}

export function calculateFundScore(data: Pick<EtfData, "netMainForce5d" | "netMainForce10d">): number {
  const { netMainForce5d = 0, netMainForce10d = 0 } = data;
  const avg = (netMainForce5d + netMainForce10d) / 2;
  // Normalize: assume +/-5000 as full range
  return Math.min(100, Math.max(0, (avg / 5000) * 50 + 50));
}

export function calculateValuationScore(data: Pick<EtfData, "pePercent" | "pbPercent">): number {
  const { pePercent = 50, pbPercent = 50 } = data;
  const avg = (pePercent + pbPercent) / 2;
  const floored = Math.max(avg, 5); // floor protection
  return Math.min(100, Math.max(0, 100 - floored));
}

export function calculateQualityScore(data: Pick<EtfData, "assetScale" | "sharpRatio1y" | "sharpRatio3y">): number {
  const { assetScale = 0, sharpRatio1y = 0, sharpRatio3y = 0 } = data;
  let score = 50;
  if (assetScale > 1e9) score += 10;
  if (sharpRatio1y > 1) score += 10;
  if (sharpRatio3y > 1) score += 5;
  return Math.min(100, Math.max(0, score));
}

export function getAdvice(score: number): string {
  if (score >= 75) return "增持";
  if (score >= 60) return "持有";
  if (score >= 45) return "减持";
  return "观望";
}

function getWeights(period: string): [number, number, number, number] {
  if (period === "short") return [0.40, 0.30, 0.15, 0.15];
  if (period === "medium") return [0.35, 0.25, 0.25, 0.15];
  return [0.25, 0.20, 0.35, 0.20];
}

function parseEtfCode(symbol: string): string | null {
  const m = symbol.match(/(\d{6})\.(SZ|SH|BJ)/);
  return m ? m[1] : null;
}

async function fetchEtfData(code: string): Promise<EtfData | null> {
  try {
    const response = await fetchGfEtfList({ tradeCode: code, limit: 1 });
    const list = response.data?.data?.fundList;
    if (!list || list.length === 0) return null;
    return list[0] as EtfData;
  } catch {
    return null;
  }
}

function calculateTotalScore(data: EtfData, period: string): number {
  const [wMomentum, wFund, wValuation, wQuality] = getWeights(period);
  const momentum = calculateMomentumScore(data, period);
  const fund = calculateFundScore(data);
  const valuation = calculateValuationScore(data);
  const quality = calculateQualityScore(data);
  return +(momentum * wMomentum + fund * wFund + valuation * wValuation + quality * wQuality).toFixed(2);
}

function formatRotationResults(
  period: string,
  results: Array<{ code: string; name: string; score: number; momentum: number; fund: number; valuation: number; quality: number; advice: string }>,
): string {
  const periodLabels: Record<string, string> = { short: "短期", medium: "中期", long: "长期" };
  const lines: string[] = [
    `## ETF 轮动策略评分（${periodLabels[period] || period}）`,
    "",
    "| 排名 | 代码 | 名称 | 综合评分 | 动量 | 资金 | 估值 | 质量 | 建议 |",
    "|------|------|------|----------|------|------|------|------|------|",
  ];

  for (let i = 0; i < results.length; i++) {
    const r = results[i];
    lines.push(`| ${i + 1} | ${r.code} | ${r.name} | ${r.score} | ${r.momentum} | ${r.fund} | ${r.valuation} | ${r.quality} | ${r.advice} |`);
  }

  lines.push("");
  lines.push("### 详细分析");
  for (const r of results) {
    lines.push(`**${r.code} | ${r.name}** — 综合评分 ${r.score}，建议${r.advice}`);
  }

  lines.push("");
  lines.push("⚠️ 不构成投资建议");
  return lines.join("\n");
}

export function createEtfRotationStrategyTool(): AnyAgentTool {
  return {
    name: "etfRotationStrategy",
    label: "ETF 轮动策略",
    description: "输入一组 ETF 代码，基于动量、资金流向、估值、质量四因子进行量化评分和轮动建议。支持 short/medium/long 三种周期权重。",
    parameters: EtfRotationStrategySchema,
    execute: async (_toolCallId, params) => {
      try {
        const p = params as Record<string, unknown>;
        const symbols = Array.isArray(p.symbols) ? p.symbols as string[] : [];
        const period = String(p.period || "").trim();
        const maxResults = Math.min(20, Math.max(1, Number(p.maxResults) || symbols.length));

        if (!["short", "medium", "long"].includes(period)) {
          return toToolResult({ content: "周期必须是 short、medium 或 long", isError: true });
        }
        if (symbols.length === 0) {
          return toToolResult({ content: "请至少提供一个 ETF 代码", isError: true });
        }

        const codes = symbols.map(parseEtfCode).filter((c): c is string => c !== null);
        if (codes.length === 0) {
          return toToolResult({ content: "所有代码格式无效，请使用 510050.SH 格式", isError: true });
        }

        const dataList = await Promise.all(codes.map(fetchEtfData));
        const results: Array<{ code: string; name: string; score: number; momentum: number; fund: number; valuation: number; quality: number; advice: string }> = [];

        for (let i = 0; i < codes.length; i++) {
          const data = dataList[i];
          if (!data) continue;
          const score = calculateTotalScore(data, period);
          results.push({
            code: codes[i],
            name: data.secuAbbr || codes[i],
            score,
            momentum: calculateMomentumScore(data, period),
            fund: calculateFundScore(data),
            valuation: calculateValuationScore(data),
            quality: calculateQualityScore(data),
            advice: getAdvice(score),
          });
        }

        if (results.length === 0) {
          return toToolResult({ content: "未能获取到任何有效 ETF 数据，请检查代码或稍后重试", isError: true });
        }

        results.sort((a, b) => b.score - a.score);
        const top = results.slice(0, maxResults);
        const output = formatRotationResults(period, top);
        return toToolResult({ content: output });
      } catch (error) {
        return toToolResult({
          content: `轮动策略执行失败: ${error instanceof Error ? error.message : String(error)}`,
          isError: true,
        });
      }
    },
  };
}
```

```ts
export function ema(data: number[], period: number): number {
  const k = 2 / (period + 1);
  let value = data.slice(0, period).reduce((a, b) => a + b, 0) / period;
  for (let i = period; i < data.length; i++) {
    value = data[i] * k + value * (1 - k);
  }
  return value;
}

export function calcMACDSeries(closes: number[]): Array<{ dif: number; dea: number; macd: number }> {
  if (closes.length < 35) return [];
  const difSeries: number[] = [];
  for (let i = 26; i <= closes.length; i++) {
    const slice = closes.slice(0, i);
    difSeries.push(ema(slice, 12) - ema(slice, 26));
  }
  const result: Array<{ dif: number; dea: number; macd: number }> = [];
  for (let i = 0; i < difSeries.length; i++) {
    const deaSlice = difSeries.slice(Math.max(0, i - 8), i + 1);
    const dea = ema(deaSlice.length >= 9 ? difSeries.slice(i - 8, i + 1) : difSeries.slice(0, i + 1), Math.min(9, deaSlice.length));
    const dif = difSeries[i];
    result.push({ dif: +dif.toFixed(2), dea: +dea.toFixed(2), macd: +((dif - dea) * 2).toFixed(2) });
  }
  return result;
}

export function detectCross(series: Array<{ dif: number; dea: number; macd: number }>): "golden" | "death" | "none" {
  if (series.length < 2) return "none";
  const prev = series[series.length - 2];
  const curr = series[series.length - 1];
  if (prev.dif <= prev.dea && curr.dif > curr.dea) return "golden";
  if (prev.dif >= prev.dea && curr.dif < curr.dea) return "death";
  return "none";
}

export function calcKDJSeries(klines: Kline[]): Array<{ k: number; d: number }> {
  if (klines.length < 9) return [];
  let k = 50;
  let d = 50;
  const result: Array<{ k: number; d: number }> = [];
  for (let i = 8; i < klines.length; i++) {
    const slice = klines.slice(i - 8, i + 1);
    const highest = Math.max(...slice.map((kk) => kk.high));
    const lowest = Math.min(...slice.map((kk) => kk.low));
    const close = klines[i].close;
    const range = highest - lowest;
    const rsv = range === 0 ? 50 : ((close - lowest) / range) * 100;
    k = (2 / 3) * k + (1 / 3) * rsv;
    d = (2 / 3) * d + (1 / 3) * k;
    result.push({ k: +k.toFixed(2), d: +d.toFixed(2) });
  }
  return result;
}
```

```ts
  lines.push(`| PE 百分位 | ${pePercent !== undefined ? pePercent.toFixed(1) + "%" : "N/A"} | ${pePercent !== undefined ? (pePercent < 30 ? "历史较低水平" : pePercent > 70 ? "历史较高水平" : "历史中位水平") : "数据缺失"} |`);
  lines.push(`| PB 百分位 | ${pbPercent !== undefined ? pbPercent.toFixed(1) + "%" : "N/A"} | ${pbPercent !== undefined ? (pbPercent < 30 ? "历史较低水平" : pbPercent > 70 ? "历史较高水平" : "历史中位水平") : "数据缺失"} |`);
  lines.push(`| 综合估值百分位 | ${compositePercent.toFixed(1)}% | ${label}区间 |`);
  lines.push(`| 指数温度 | ${data.indexTempType || "N/A"} | 广发: ${data.indexTempType === "low" ? "低温" : data.indexTempType === "ord" ? "中温" : data.indexTempType === "high" ? "高温" : "未知"} |`);

  lines.push("");
  lines.push("**定投建议：**");
  lines.push(`- 基础金额：${baseAmount} 元`);
  lines.push(`- 建议倍数：${multiplier.toFixed(1)}x`);
  lines.push(`- 本次建议投入：${amount} 元`);

  if (conflictWarning) {
    lines.push("");
    lines.push(`⚠️ ${conflictWarning}`);
  }

  lines.push("");
  lines.push(`**策略逻辑：** 综合估值百分位 ${compositePercent.toFixed(1)}%，处于${label}区间，${multiplier >= 1.5 ? "建议加倍定投以积累更多低成本份额" : multiplier >= 1 ? "按标准金额定投" : multiplier > 0 ? "建议减少定投金额" : "建议暂停定投"}。`);
  lines.push("⚠️ 不构成投资建议");

  return lines.join("\n");
}

export function createEtfSmartInvestTool(): AnyAgentTool {
  return {
    name: "etfSmartInvest",
    label: "ETF 智能定投",
    description: "根据 ETF 当前估值温度（PE/PB 百分位）给出智能定投倍数建议。输入基础金额，输出建议投入金额。",
    parameters: EtfSmartInvestSchema,
    execute: async (_toolCallId, params) => {
      try {
        const p = params as Record<string, unknown>;
        const symbol = String(p.symbol || "").trim();
        const baseAmount = Math.max(1, Number(p.baseAmount) || 1000);

        const code = parseEtfCode(symbol);
        if (!code) {
          return toToolResult({ content: "代码格式无效，请使用 510050.SH 格式", isError: true });
        }

        const data = await fetchValuationData(code);
        if (!data) {
          return toToolResult({ content: "未能获取到该 ETF 的估值数据，请检查代码或稍后重试", isError: true });
        }

        const pePercent = data.pePercent ?? undefined;
        const pbPercent = data.pbPercent ?? undefined;
        const tempType = data.indexTempType ?? undefined;

        let compositePercent: number | undefined;
        let multiplier: number;
        let conflictWarning: string | null = null;

        if (pePercent !== undefined || pbPercent !== undefined) {
          const available: number[] = [];
          if (pePercent !== undefined) available.push(pePercent);
          if (pbPercent !== undefined) available.push(pbPercent);
          compositePercent = available.reduce((a, b) => a + b, 0) / available.length;
          multiplier = calculateMultiplier(compositePercent);

          // Conflict detection
          if (tempType) {
            const tempMultiplier = tempType === "low" ? 1.5 : tempType === "ord" ? 1.0 : 0.0;
            if (Math.abs(multiplier - tempMultiplier) > 1.0) {
              conflictWarning = `数据不一致：PE/PB 百分位建议 ${multiplier}x，但广发指数温度为 ${tempType}，请综合判断。`;
            }
          }
        } else if (tempType) {
          // Fallback to indexTempType only
          compositePercent = tempType === "low" ? 20 : tempType === "ord" ? 50 : 80;
          multiplier = calculateMultiplier(compositePercent);
        } else {
          return toToolResult({ content: "该 ETF 缺乏估值数据（PE/PB 百分位和指数温度均不可用），无法给出定投建议", isError: true });
        }

        const output = formatInvestOutput(data, baseAmount, pePercent, pbPercent, compositePercent!, multiplier, conflictWarning);
        return toToolResult({ content: output });
      } catch (error) {
        return toToolResult({
          content: `智能定投建议失败: ${error instanceof Error ? error.message : String(error)}`,
          isError: true,
        });
      }
    },
  };
}
```
